```Bash
# Install cbt
gcloud components install cbt

# Create an instance
cbt createinstance

# Create a cluster within a configured instance
cbt createcluster

# Create a table
cbt createtable

# Delete an instance
cbt deleteinstance

# Delete a cluster
cbt deletecluster

# Delete a table
cbt deletetable
```

```Bash
# List all cbt instances
cbt listinstances

# List all tables and column families
cbt ls
```
